#include <iostream>
#include <stdexcept> // Nagłówek do wyjątków standardowych

// Funkcja dzieląca dwie liczby
double divide(double numerator, double denominator) {
    if (denominator == 0) {
        // Jeśli dzielnik to 0, zgłaszamy wyjątek
        throw std::invalid_argument("Blad: Proba dzielenia przez zero!");
    }
    return numerator / denominator;
}

int main() {
    double num1, num2;

    // Pobieranie danych od użytkownika
    std::cout << "Podaj licznik: ";
    std::cin >> num1;
    std::cout << "Podaj mianownik: ";
    std::cin >> num2;

    try {
        // Próbujemy wykonać dzielenie
        double result = divide(num1, num2);
        std::cout << "Wynik dzielenia: " << result << std::endl;
    }
    catch (const std::invalid_argument& e) {
        // Obsługa wyjątku, gdy dzielnik wynosi zero
        std::cout << e.what() << std::endl;
    }

    return 0;
}
